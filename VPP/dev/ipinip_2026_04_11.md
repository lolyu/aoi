## valid packet forwarding
```
from scapy.all import *

# With ICMP inner
pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IP(src="10.20.20.10", dst="10.1.0.32") /
       IP(src="10.10.10.10", dst="20.20.20.20") /
       ICMP())

pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IP(src="10.20.20.10", dst="10.0.0.60") /
       IP(src="10.10.10.10", dst="20.20.20.20") /
       ICMP())

pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IP(src="10.20.20.10", dst="192.168.0.1") /
       IP(src="10.10.10.10", dst="20.20.20.20") /
       ICMP())

pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IP(src="10.20.20.10", dst="192.168.0.1") /
       IP(src="10.10.10.11", dst="20.20.20.20") /
       ICMP())

# Or with UDP inner
pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IP(src="10.10.10.10", dst="192.168.0.1") /
       IP(src="10.10.10.10", dst="20.20.20.20") /
       UDP(sport=1234, dport=5678) /
       Raw(b"D" * 64))

sendp(pkt, iface="eth28", count=10000)

# IPv6 outer, IPv6 inner (IP6inIP6)
pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IPv6(src="fc00::1", dst="fc00:1::32") /
       IPv6(src="2001:db8::1", dst="2001:db8::2") /
       ICMPv6EchoRequest())

pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IPv6(src="fc00::1", dst="fc00::75") /
       IPv6(src="2001:db8::1", dst="2001:db8::2") /
       ICMPv6EchoRequest())

pkt = (Ether(dst="22:59:d0:d5:19:a9", src="3e:38:be:ea:b0:1c") /
       IPv6(src="fc00::1", dst="fc02:1000::1") /
       IPv6(src="2001:db8::1", dst="2001:db8::2") /
       ICMPv6EchoRequest())


sendp(pkt, iface="eth28", count=10000)
```
