```bash
# Inside the syncd container, replace RELP with UDP forwarding:
docker exec syncd bash -c '
  sed -i "s|module(load=\"omrelp\")|#module(load=\"omrelp\")|" /etc/rsyslog.conf
  sed -i "s|.*action(type=\"omrelp\".*|*.* action(type=\"omfwd\" target=\"127.0.0.1\" port=\"514\" protocol=\"udp\" Template=\"ForwardFormatInContainer\")|" /etc/rsyslog.conf
  supervisorctl restart rsyslogd
'
```
